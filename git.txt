git init  初始化
git status 查看文件状态
git add 文件  上传
git commit -m '版本' 保存版本
git log 查看版本
git reflog  回滚后查看版本号
git config user.name“用户名”绑定用户名
git config user.email “邮箱” 绑定邮箱
git reset --hard 'commit' /开头   回滚
git checkout master/分支名 切换分支
git merge 分支名 合并分支到主干
git branch 分支名称 创建分支
git branch -d 分支名 删除bug分支
